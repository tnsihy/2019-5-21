$("ul li:first-child") | 选取每个`<ul>`元素的第一个`<li>`元素
$("[href]") | 选取带有href属性的元素
$("a[target='_blank']") | 选取所有target属性值等于 "_blank"的`<a>`元素
$("a[target!='_blank']") | 
$(":button") | 选取所有type="button"的`<input>`元素或者`<button>元素`
$("tr:even") | 选取偶数位置的`<tr>`元素
$("tr:odd") | 选取奇数位置的`<tr>`元素
